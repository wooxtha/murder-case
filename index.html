""" Relationship Map: Murder at Tanjong Pagar Law Firm -------------------------------------------------- This script visualizes the relationships between individuals involved in the fictional murder case using NetworkX and Matplotlib. """ import networkx as nx import matplotlib.pyplot as plt # Create an undirected graph G = nx.Graph() # Define all individuals people = [ "Christina Lee (Victim)", "Stephen Glisson", "Michelle Ho", "Clarence Cheung", "Jason Lee", "Sophia Ibrahim", "Lilyan Lam", "Simon Carter", "Bo Jing", "Tan Choon Beng", "Charles Park" ] # Add individuals as nodes G.add_nodes_from(people) # Define known relationships between people relationships = [ ("Christina Lee (Victim)", "Stephen Glisson"), # Law firm partners ("Christina Lee (Victim)", "Michelle Ho"), # Law firm partners ("Stephen Glisson", "Bo Jing"), # Client meeting ("Michelle Ho", "Simon Carter"), # Client meeting ("Michelle Ho", "Charles Park"), # Married ("Clarence Cheung", "Stephen Glisson"), # Asked for promotion ("Clarence Cheung", "Bo Jing"), # Drafted will ("Clarence Cheung", "Tan Choon Beng"), # Side job and dinner ("Jason Lee", "Christina Lee (Victim)"), # Cousins ("Jason Lee", "Sophia Ibrahim"), # Worked together ("Jason Lee", "Clarence Cheung"), # Co-workers ("Sophia Ibrahim", "Christina Lee (Victim)"), # Conflict incident ("Lilyan Lam", "Christina Lee (Victim)"), # Found the body ("Tan Choon Beng", "Charles Park"), # Side business ] # Add relationships as edges in the graph G.add_edges_from(relationships) # Draw the graph plt.figure(figsize=(16, 10)) pos = nx.spring_layout(G, seed=42) # Seed for consistent layout # Node styling nx.draw_networkx_nodes(G, pos, node_color="#66b3ff", node_size=1800, edgecolors='black') # Edge styling nx.draw_networkx_edges(G, pos, edge_color="gray", width=2) # Label styling nx.draw_networkx_labels(G, pos, font_size=9, font_family="sans-serif", font_weight="bold") # Title and layout plt.title("Relationship Map: Murder at Tanjong Pagar Law Firm", fontsize=14, fontweight='bold') plt.axis("off") plt.tight_layout() plt.show()
